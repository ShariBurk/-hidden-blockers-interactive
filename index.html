async function startListening(bi, qi) {

  const micBtn  = document.getElementById('mic-' + bi + '-' + qi);

  const statusEl = document.getElementById('mic-status-' + bi + '-' + qi);

  let stream;

  try {

    stream = await navigator.mediaDevices.getUserMedia({ audio: true });

  } catch(e) {

    if (statusEl) statusEl.innerHTML = 'Microphone access denied. Please choose an option below.';

    return;

  }

  audioChunks = [];

  currentlyListening = true;

  if (micBtn) { micBtn.classList.add('listening'); micBtn.style.background='#d0021b'; }

  if (statusEl) statusEl.innerHTML = '<strong>Listening…</strong> Speak naturally, then click mic again to stop.';


  const mimeType = MediaRecorder.isTypeSupported('audio/webm') ? 'audio/webm' : 'audio/mp4';

  mediaRecorder = new MediaRecorder(stream, { mimeType });

  mediaRecorder.ondataavailable = e => { if (e.data.size > 0) audioChunks.push(e.data); };

  mediaRecorder.onstop = async () => {

    stream.getTracks().forEach(t => t.stop());

    if (statusEl) statusEl.innerHTML = 'Transcribing…';

    await transcribeWhisper(bi, qi, new Blob(audioChunks, { type: mimeType }), mimeType);

  };

  mediaRecorder.start();

}

  const micBtn = (bi !== undefined) ? document.getElementById('mic-' + bi + '-' + qi) : null;

  if (micBtn) { micBtn.classList.remove('listening'); micBtn.style.background='#00A3D8'; }

}

    const res = await fetch('https://api.openai.com/v1/audio/transcriptions', {

    const data = await res.json();

function goToPrevBlocker(i) {

  const idx = S.relevantBlockers.indexOf(i);

  if (idx > 0) showBlocker(S.relevantBlockers[idx - 1]);

}


function nextBlocker(i) {

  S.completed[i] = true;

  updateSidebar();

  

  const idx = S.relevantBlockers.indexOf(i);

  if (idx !== -1 && idx < S.relevantBlockers.length - 1) {

    showBlocker(S.relevantBlockers[idx + 1]);

  } else {

    

    showReport();

  }

}


function updateSidebar() {

  S.relevantBlockers.forEach(i => {

    const el = document.getElementById('sb-' + i);

    if (!el) return;

    el.className = 'sb-item';

    if (i === S.current) el.classList.add('active');

    if (S.completed[i]) el.classList.add('done');

  });


  

  const allDone = S.relevantBlockers.length > 0 &&

    S.relevantBlockers.every(i => S.completed[i]);

  const btn = document.getElementById('sbReportBtn');

  if (btn) btn.style.display = allDone ? 'block' : 'none';

}


function updateProgress() {

  if (!S.relevantBlockers.length) return;

  const totalQ = S.relevantBlockers.length * 2;

  const answered = S.relevantBlockers.reduce((acc, i) =>

    acc + S.answers[i].filter(a => a !== null).length, 0);

  const pct = Math.round((answered / totalQ) * 100);

  document.getElementById('tbFill').style.width = pct + '%';

  document.getElementById('tbPct').textContent = pct + '%';

}


function populateCalcScores() {

  const container = document.getElementById('calcScoresSummary');

  if (!container) return;

  const order = {Critical:4,High:3,Medium:2,Low:1,Skipped:0};

  const scored = BLOCKERS.map((b,i) => ({name:b.name, sev:getSeverity(i)}))

    .sort((a,b) => order[b.sev] - order[a.sev]);

      <div class="calc-score-val ${s.sev.toLowerCase()}">${s.sev}</div>

    </div>`).join('');

}

function getSeverity(bi) {

  const answers = S.answers[bi].filter(a=>a!==null);

  if (!answers.length) return 'Skipped';

  const map = {Low:1,Medium:2,High:3,Critical:4};

  const avg = answers.reduce((s,a)=>s+map[a.sev],0)/answers.length;

  if (avg<=1.5) return 'Low';

  if (avg<=2.5) return 'Medium';

  if (avg<=3.5) return 'High';

  return 'Critical';

}
